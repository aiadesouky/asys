DECLARE
   STAT   BOOLEAN;
BEGIN
   DBMS_OUTPUT.DISABLE;
   DBMS_OUTPUT.ENABLE (100000);
   STAT := FND_PROFILE.SAVE ('FND_DISABLE_OA_CUSTOMIZATIONS', 'Y', 'SITE');

   IF STAT
   THEN
      DBMS_OUTPUT.PUT_LINE ('PROFILE UPDATED');
   ELSE
      DBMS_OUTPUT.PUT_LINE ('PROFILE NOT UPDATED');
   END IF;

   COMMIT;
END;
